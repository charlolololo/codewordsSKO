        // cred: chatgpt
        // Intialise variables
        let images = [];  // Declare the array to hold images
        let imgSize = 200;  // Size of each image
        let cols, rows;
        let gridHeight, gridWidth;
        let startX, startY;

        // Add pictures to image array
        function preload() {
            // Load images directly into the array in one line
            images = [
            loadImage('data/1_giraffe.jpg'),
            loadImage('data/2_starfish.jpg'),
            loadImage('data/3_fluffy_octo.jpg'),
            loadImage('data/4_angry_frog.jpg'),
            loadImage('data/5_angry_turtle.jpg'),
            loadImage('data/6_aubergine.jpg'),
            loadImage('data/7_sheepie.jpg'),
            loadImage('data/8_brontosaurus.jpg'),
            loadImage('data/9_octo.jpg')
            ];
        }

        function setup() {
            createCanvas(windowWidth, windowHeight);
            cols = 3;  // Number of columns in the grid
            rows = 3;  // Number of rows in the grid
  
            gridWidth = cols * imgSize;
            gridHeight = rows * imgSize;
  
            startX = (width - gridWidth) / 2;
            startY = (height - gridHeight) / 2;
        }

        function draw() {
        background(255);
        imageMode(CORNER);
        let index = 0;

        // Nested loops to display the images in a grid
        for (let row = 0; row < rows; row++) {
            for (let col = 0; col < cols; col++) {
            if (index < images.length) {
                let img = images[index];
                let x = startX + col * imgSize;
                let y = startY + row * imgSize;

                // Calculate proportional size to maintain aspect ratio
                let aspectRatio = img.width / img.height;
                let w, h;
                
                if (aspectRatio > 1) {  // Landscape image (wider than tall)
                w = imgSize;
                h = imgSize / aspectRatio;
                } else {  // Portrait image (taller than wide) or square
                h = imgSize;
                w = imgSize * aspectRatio;
                }

                // Center the image within the grid cell
                let xOffset = (imgSize - w) / 2;
                let yOffset = (imgSize - h) / 2;

                // Draw the image at the centered position
                image(img, x + xOffset, y + yOffset, w, h);
                index++;
            }
            }
        }
